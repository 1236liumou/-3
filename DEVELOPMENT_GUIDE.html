车牌识别系统 — 完整开发指南

数字图像处理课程设计 · Python + OpenCV · 端到端流水线

项目简介

本项目实现了一个端到端的车牌识别系统，覆盖从图像输入到车牌号输出的完整流水线。系统分为五个核心阶段：图像预处理 → 车牌定位 → 倾斜校正 → 字符分割 → 字符识别，每个阶段使用经典的数字图像处理算法实现。

车牌类型蓝牌、绿牌（新能源）、黄牌

数据集CCPD（30 万+ 中国车牌图像，开源）
界面Streamlit 交互式可视化，展示每阶段中间结果

Step 2工具函数 utils.py

提供跨模块共享的工具函数，包括非侵入式计时装饰器、图像格式转换、透视变换、NMS、结果绘制和车牌格式校验。

lpr/utils.pyimport cv2
import numpy as np
import time
from functools import wraps


def timer(func):
    """非侵入式计时装饰器，时间存储在实例的 _timings 字典中"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = (time.perf_counter() - start) * 1000
        if args and hasattr(args[0], '_timings'):
            args[0]._timings[func.__name__] = elapsed
        return result
    return wrapper


def to_gray(image):
    """BGR 转灰度图"""
    if len(image.shape) == 3:
        return cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    return image.copy()


def to_hsv(image):
    """BGR 转 HSV"""
    return cv2.cvtColor(image, cv2.COLOR_BGR2HSV)


def order_points(pts):
    """对四点排序: [左上, 右上, 右下, 左下]"""
    rect = np.zeros((4, 2), dtype="float32")
    s = pts.sum(axis=1)
    rect[0] = pts[np.argmin(s)]   # 左上: 和最小
    rect[2] = pts[np.argmax(s)]   # 右下: 和最大
    diff = np.diff(pts, axis=1)
    rect[1] = pts[np.argmin(diff)]  # 右上: 差最小
    rect[3] = pts[np.argmax(diff)]  # 左下: 差最大
    return rect


def four_point_transform(image, pts):
    """四点透视变换，将倾斜车牌校正为矩形"""
    rect = order_points(pts)
    tl, tr, br, bl = rect
    # 计算新图像宽高
    width_a = np.sqrt((br[0]-bl[0])**2 + (br[1]-bl[1])**2)
    width_b = np.sqrt((tr[0]-tl[0])**2 + (tr[1]-tl[1])**2)
    max_width = max(int(width_a), int(width_b))
    height_a = np.sqrt((tr[0]-br[0])**2 + (tr[1]-br[1])**2)
    height_b = np.sqrt((tl[0]-bl[0])**2 + (tl[1]-bl[1])**2)
    max_height = max(int(height_a), int(height_b))

    dst = np.array([
        [0, 0], [max_width-1, 0],
        [max_width-1, max_height-1], [0, max_height-1]
    ], dtype="float32")

    m = cv2.getPerspectiveTransform(rect, dst)
    return cv2.warpPerspective(image, m, (max_width, max_height))


def validate_plate_format(plate_text):
    """车牌格式校验: 省份(1) + 字母(1) + 字母或数字(5~6)"""
    if not plate_text or len(plate_text) < 7:
        return False
    provinces = list("京津沪渝冀晋蒙辽吉黑苏浙皖闽赣鲁豫鄂湘粤桂琼川贵云藏陕甘青宁新")
    if plate_text[0] not in provinces:
        return False
    letters = "ABCDEFGHJKLMNPQRSTUVWXYZ"
    if plate_text[1] not in letters:
        return False
    valid = "0123456789ABCDEFGHJKLMNPQRSTUVWXYZ"
    return all(c in valid for c in plate_text[2:])
关键设计：非侵入式计时装饰器
早期版本的 @timer 会修改函数返回值（追加 elapsed 到 tuple），导致调用链解包全部出错。改进后的版本不修改返回值，而是将计时结果存入实例的 _timings 字典，干净利落。

Step 3图像预处理 preprocess.py

Stage 1 流程

输入: 原始彩色车辆图像 → 输出: 增强后的灰度图 + 原始彩色图（供颜色定位使用）

  1. 灰度化 — Y = 0.299R + 0.587G + 0.114B，为后续滤波做准备
  2. 双边滤波 — 保边去噪，同时考虑空间距离和像素值差异，保留车牌边缘
  3. CLAHE 增强 — 限制对比度自适应直方图均衡化，改善低照度/低对比度图像
lpr/preprocess.pyimport cv2
import numpy as np
from .config import PREPROCESS_CONFIG
from .utils import timer, to_gray


class ImagePreprocessor:
    """图像预处理器"""

    def __init__(self, config=None):
        self.config = config or PREPROCESS_CONFIG
        self._timings = {}

    @timer
    def grayscale(self, image):
        """灰度化: RGB -> Gray"""
        return to_gray(image)

    @timer
    def bilateral_filter(self, image):
        """
        双边滤波 (保边去噪)
        - 空间高斯核: 距离越近权重越大
        - 值域高斯核: 像素值越接近权重越大
        两者相乘实现"在平坦区域去噪、在边缘保留"的效果
        """
        return cv2.bilateralFilter(
            image,
            self.config["bilateral_d"],          # 邻域直径
            self.config["bilateral_sigma_color"],  # 颜色空间标准差
            self.config["bilateral_sigma_space"],  # 坐标空间标准差
        )

    @timer
    def clahe_enhance(self, gray_image):
        """
        CLAHE: 限制对比度自适应直方图均衡化
        - 将图像分块 (tile)，每块独立做直方图均衡
        - 限制对比度增强幅度 (clipLimit)，防止噪声放大
        - 块间双线性插值，消除块边界
        """
        clahe = cv2.createCLAHE(
            clipLimit=self.config["clahe_clip_limit"],
            tileGridSize=self.config["clahe_grid_size"],
        )
        return clahe.apply(gray_image)

    def process(self, image):
        """完整预处理流水线"""
        self._timings = {}
        color = image.copy()
        gray = self.grayscale(image)
        denoised = self.bilateral_filter(gray)
        enhanced = self.clahe_enhance(denoised)
        return {
            "gray": gray,
            "enhanced": enhanced,
            "denoised": denoised,
            "color": color,
            "timing": dict(self._timings),
        }

Step 4车牌定位 plate_locator.py

Stage 2 流程 (颜色法)

输入: 彩色图 + 灰度图 → 输出: 车牌区域坐标

  1. HSV 颜色过滤 — BGR → HSV，用 inRange() 提取蓝/绿/黄色区域
  2. 形态学闭操作 — 用宽矩形核 (25×10) 连接字符区域形成连通块
  3. 轮廓检测 — findContours() 找到所有外轮廓
  4. 候选筛选 — 面积、宽高比 (1.5~6.0)、矩形度 三重条件过滤
  5. 评分选择 — 宽高比接近标准值 3.14 + 矩形度 + 面积 综合评分
lpr/plate_locator.pyimport cv2
import numpy as np
from .config import PLATE_LOCATOR_CONFIG
from .utils import timer, to_hsv


class PlateLocator:
    """车牌定位器"""

    def __init__(self, config=None):
        self.config = config or PLATE_LOCATOR_CONFIG
        self._timings = {}

    @timer
    def color_filter(self, hsv_image, plate_type="blue"):
        """
        HSV 颜色空间过滤
        蓝牌: H ∈ [100,124], 绿牌: H ∈ [35,99], 黄牌: H ∈ [15,34]
        """
        lower = np.array(self.config[f"{plate_type}_lower"])
        upper = np.array(self.config[f"{plate_type}_upper"])
        mask = cv2.inRange(hsv_image, lower, upper)
        # 形态学开操作去噪
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        return mask

    @timer
    def morphological_close(self, binary_image):
        """
        形态学闭操作: 膨胀 -> 腐蚀
        连接相近区域，填充孔洞，形成连通块
        宽矩形核 (25x10) 水平方向连接字符
        """
        kernel = cv2.getStructuringElement(
            cv2.MORPH_RECT, self.config["morph_kernel_size"]
        )
        return cv2.morphologyEx(
            binary_image, cv2.MORPH_CLOSE, kernel,
            iterations=self.config["morph_iterations"]
        )

    @timer
    def find_plate_candidates(self, binary_image):
        """
        轮廓检测 + 三重筛选:
        1. 面积在 [1500, 50000] 范围内
        2. 宽高比在 [1.5, 6.0] 范围内
        3. 矩形度 (轮廓面积/外接矩形面积) > 0.3
        """
        contours, _ = cv2.findContours(
            binary_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
        )
        candidates = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if not (self.config["min_plate_area"] < area < self.config["max_plate_area"]):
                continue
            rect = cv2.minAreaRect(contour)
            (cx, cy), (w, h), angle = rect
            if w < h: w, h = h, w
            aspect_ratio = w / h if h > 0 else 0
            if not (self.config["min_aspect_ratio"] < aspect_ratio < self.config["max_aspect_ratio"]):
                continue
            rect_score = area / (w * h) if w * h > 0 else 0
            if rect_score < self.config["min_rect_score"]:
                continue
            box = cv2.boxPoints(rect)
            x, y, rw, rh = cv2.boundingRect(contour)
            candidates.append({
                "box": np.int0(box), "rect": (x, y, rw, rh),
                "center": (int(cx), int(cy)), "size": (int(w), int(h)),
                "angle": angle, "area": area,
                "aspect_ratio": aspect_ratio, "rect_score": rect_score,
            })
        return candidates

    @timer
    def select_best_candidate(self, candidates):
        """
        综合评分: 宽高比接近度(0.4) + 矩形度(0.3) + 面积(0.3)
        标准车牌宽高比 = 440mm / 140mm ≈ 3.14
        """
        if not candidates:
            return None
        best_score, best = -1, None
        target_ratio = 3.14
        for cand in candidates:
            ratio_score = 1.0 - abs(cand["aspect_ratio"] - target_ratio) / target_ratio
            area_score = min(cand["area"] / self.config["max_plate_area"], 1.0)
            score = ratio_score * 0.4 + cand["rect_score"] * 0.3 + area_score * 0.3
            cand["score"] = score
            if score > best_score:
                best_score, best = score, cand
        return best

    def locate(self, color_image, gray_image, method="color"):
        """车牌定位主函数"""
        self._timings = {}
        if method in ("color", "multi"):
            hsv = to_hsv(color_image)
            mask = self.color_filter_multi(hsv) if method == "multi" else self.color_filter(hsv, "blue")
        else:
            mask = self.sobel_edge(gray_image)
        closed = self.morphological_close(mask)
        candidates = self.find_plate_candidates(closed)
        best = self.select_best_candidate(candidates)
        return {
            "plate_region": best,
            "all_candidates": candidates,
            "binary": closed,
            "method": method,
            "timing": dict(self._timings),
        }
为什么用 HSV 而不是 RGB？
RGB 颜色空间中，颜色和亮度耦合在一起，同一蓝色在不同光照下 RGB 值差异很大。HSV 空间将色相 (H) 与亮度 (V) 分离，蓝色始终对应 H ∈ [100, 124]，不受光照变化影响。

Step 5倾斜校正 skew_correction.py

Stage 3 流程

输入: 裁剪的车牌图像 → 输出: 水平校正后的车牌图像

  1. Canny 边缘检测 — 提取车牌区域的边缘
  2. Hough 直线检测 — 检测车牌上下边框线
  3. 角度统计 — 过滤近水平线 (|angle| < 45°)，取中位数作为倾斜角
  4. 仿射变换 — getRotationMatrix2D + warpAffine 旋转校正
lpr/skew_correction.pyimport cv2
import numpy as np
from .config import SKEW_CORRECTION_CONFIG
from .utils import timer, four_point_transform


class SkewCorrector:
    """车牌倾斜校正器"""

    def __init__(self, config=None):
        self.config = config or SKEW_CORRECTION_CONFIG
        self._timings = {}

    @timer
    def correct_by_hough(self, plate_image, plate_region=None):
        """
        Hough 直线检测法倾斜校正
        1. Canny 边缘检测
        2. HoughLinesP 检测直线
        3. 统计近水平线的角度，取中位数
        4. 仿射变换旋转
        """
        gray = cv2.cvtColor(plate_image, cv2.COLOR_BGR2GRAY) if len(plate_image.shape) == 3 else plate_image.copy()
        h, w = gray.shape[:2]
        edges = cv2.Canny(gray, 50, 150)
        lines = cv2.HoughLinesP(edges, 1, np.pi/180,
                                threshold=30, minLineLength=20, maxLineGap=5)
        angles = []
        if lines is not None:
            for line in lines:
                x1, y1, x2, y2 = line[0]
                angle = np.degrees(np.arctan2(y2-y1, x2-x1))
                if abs(angle) < 45 or abs(angle) > 135:
                    if angle > 90: angle -= 180
                    elif angle < -90: angle += 180
                    angles.append(angle)
        skew_angle = np.median(angles) if angles else 0
        skew_angle = np.clip(skew_angle, -20, 20)
        # 仿射变换旋转
        M = cv2.getRotationMatrix2D((w/2, h/2), skew_angle, 1.0)
        corrected = cv2.warpAffine(plate_image, M, (w, h),
                                   flags=cv2.INTER_CUBIC,
                                   borderMode=cv2.BORDER_REPLICATE)
        return corrected, skew_angle

    def correct(self, plate_image, plate_region=None, method="hough"):
        """倾斜校正主函数"""
        self._timings = {}
        if not self.config["enable"]:
            return plate_image, 0, {}
        if method == "perspective" and plate_region:
            corrected, angle = self.correct_by_perspective(plate_image, plate_region)
        else:
            corrected, angle = self.correct_by_hough(plate_image, plate_region)
        return corrected, angle, dict(self._timings)

Step 6字符分割 char_segmentation.py

Stage 4 流程

输入: 校正后的车牌图像 → 输出: 7 个标准化的字符图像 (20×28)

  1. 车牌标准化 — 缩放到固定尺寸 240×80，保证后续参数一致
  2. 去除边框 — 裁剪边缘 8%，去除蓝/黄边框干扰
  3. Otsu 二值化 — 自动阈值二值化，统一为黑底白字
  4. 去噪 — 形态学闭操作 + 连通域分析去小噪声
  5. 垂直投影分割 — 统计每列白像素数，波谷处切割
  6. 字符标准化 — 保持比例缩放到 20×28，居中放置
lpr/char_segmentation.py (核心方法)@timer
def segment_by_projection(self, binary_image):
    """
    垂直投影法字符分割

    原理:
    1. 统计每列白像素总数 → 投影曲线
    2. 投影值高 = 字符 (波峰)
    3. 投影值低 = 字符间隙 (波谷)
    4. 在波谷处切割

         投影图示意:
         |  ██  █  ██   █  █  ██
         |  ██  █  ██   █  █  ██
         |________________________
          京  A  1   2  3  4  5
    """
    h, w = binary_image.shape
    # 1. 垂直投影: 每列白像素数
    projection = np.sum(binary_image > 0, axis=0).astype(np.float32)

    # 2. 平滑 (减少噪声导致的误分割)
    projection = cv2.GaussianBlur(
        projection.reshape(1, -1), (5, 1), 0
    ).flatten()

    # 3. 动态阈值
    threshold = np.max(projection) * self.config["projection_threshold_ratio"]

    # 4. 寻找连续字符区域
    in_char = False
    char_regions = []
    for x in range(w):
        if projection[x] > threshold:
            if not in_char:
                in_char = True
                start = x
        else:
            if in_char:
                in_char = False
                if x - start >= self.config["min_char_width"]:
                    char_regions.append((start, x))

    # 5. 分割并处理粘连字符
    char_images, char_boxes = [], []
    for start, end in char_regions:
        char_img = binary_image[:, start:end]
        width = end - start
        if width > self.config["max_char_width"]:
            # 粘连: 从中间切割
            mid = width // 2
            char_images.append(char_img[:, :mid])
            char_images.append(char_img[:, mid:])
        else:
            char_images.append(char_img)

    return char_images, char_boxes


@timer
def standardize_char(self, char_image):
    """
    字符标准化: 保持宽高比缩放到 20x28，黑色填充居中
    """
    target_w, target_h = self.config["char_resize"]  # (20, 28)
    h, w = char_image.shape[:2]
    scale = min(target_w / w, target_h / h)
    new_w, new_h = int(w * scale), int(h * scale)
    resized = cv2.resize(char_image, (new_w, new_h), interpolation=cv2.INTER_CUBIC)
    canvas = np.zeros((target_h, target_w), dtype=np.uint8)
    x_off = (target_w - new_w) // 2
    y_off = (target_h - new_h) // 2
    canvas[y_off:y_off+new_h, x_off:x_off+new_w] = resized
    return canvas

Step 7字符识别 char_recognition.py

三种识别方案对比
方案原理训练数据准确率速度适合场景
模板匹配cv2.matchTemplate 归一化互相关无需较低 (60-70%)快开箱即用，课程演示

lpr/char_recognition.py (核心方法)class CharRecognizer:
    """字符识别器: 模板匹配 / SVM / CNN 三选一"""

    def __init__(self, config=None):
        self.config = config or CHAR_RECOGNITION_CONFIG
        self.method = self.config["method"]
        self._timings = {}
        self._load_models()

    @timer
    def recognize_template(self, char_image, position=0):
        """
        模板匹配识别
        - position=0: 省份汉字候选集 (31 个)
        - position=1: 字母候选集 (24 个)
        - position≥2: 字母+数字候选集 (34 个)
        """
        _, char_image = cv2.threshold(char_image, 127, 255, cv2.THRESH_BINARY)

        # 根据位置选择候选集
        if position == 0:
            candidates = self.templates.get("provinces", {})
        elif position == 1:
            candidates = self.templates.get("letters", {})
        else:
            candidates = {**self.templates.get("letters", {}),
                          **self.templates.get("digits", {})}

        # 遍历模板，找最佳匹配
        best_char, best_score = "?", -1
        method = eval(self.config["template_match_method"])
        for char, template in candidates.items():
            if char_image.shape != template.shape:
                template = cv2.resize(template, (char_image.shape[1], char_image.shape[0]))
            score = float(cv2.matchTemplate(char_image, template, method)[0, 0])
            if score > best_score:
                best_score, best_char = score, char
        return best_char, max(0.0, min(1.0, best_score))

    def _extract_features(self, char_image):
        """
        HOG + 像素特征 (用于 SVM)
        HOG: 方向梯度直方图，捕捉字符形状
        像素: 10x14 降采样，补充全局信息
        """
        if char_image.shape != (28, 20):
            char_image = cv2.resize(char_image, (20, 28))
        hog = cv2.HOGDescriptor(
            _winSize=(20, 28), _blockSize=(10, 14),
            _blockStride=(5, 7), _cellSize=(5, 7), _nbins=9,
        )
        hog_features = hog.compute(char_image).flatten()
        pixel_features = cv2.resize(char_image, (10, 14)).flatten().astype(np.float32) / 255.0
        return np.concatenate([hog_features, pixel_features])

    def recognize_all(self, char_images):
        """批量识别字符 → 车牌字符串"""
        import time
        results, details, confidences = [], [], []
        for i, char_img in enumerate(char_images):
            start = time.perf_counter()
            char, conf = self.recognize(char_img, position=i)
            elapsed = (time.perf_counter() - start) * 1000
            results.append(char)
            confidences.append(conf)
            details.append({"position": i, "char": char,
                           "confidence": conf, "timing": elapsed})
        plate_text = "".join(results)
        avg_conf = float(np.mean(confidences)) if confidences else 0.0
        return plate_text, details, avg_conf, {}


class SimpleCharCNN(nn.Module):
    """
    CNN 字符识别网络
    输入: 3×28×20 (RGB)  输出: 65 类
    结构: Conv(32) → Pool → Conv(64) → Pool → FC(128) → FC(65)
    """
    def __init__(self, num_classes=65):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 32, kernel_size=3, padding=1), nn.ReLU(),
            nn.MaxPool2d(2),  # 14×10
            nn.Conv2d(32, 64, kernel_size=3, padding=1), nn.ReLU(),
            nn.MaxPool2d(2),  # 7×5
        )
        self.classifier = nn.Sequential(
            nn.Linear(64*7*5, 128), nn.ReLU(), nn.Dropout(0.5),
            nn.Linear(128, num_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        return self.classifier(x)

Step 8流水线集成 pipeline.py

将五个阶段串联为端到端流水线，统一管理错误处理、计时统计和结果汇总。

lpr/pipeline.py (核心流程)class LPRPipeline:
    """车牌识别流水线"""

    def __init__(self, locate_method="color", segment_method="projection",
                 recognize_method="template", verbose=False):
        self.preprocessor = ImagePreprocessor()
        self.locator = PlateLocator()
        self.corrector = SkewCorrector()
        self.segmenter = CharSegmenter()
        config = CHAR_RECOGNITION_CONFIG.copy()
        config["method"] = recognize_method
        self.recognizer = CharRecognizer(config=config)
        self.locate_method = locate_method
        self.segment_method = segment_method

    def recognize(self, image_input):
        """端到端车牌识别"""
        total_start = time.perf_counter()
        result = {"success": False, "plate_text": "", ...}

        # 加载图像
        image = cv2.imread(image_input) if isinstance(image_input, str) else image_input.copy()
        if image.shape[1] > 1280:  # 缩放过大图像
            image = resize_with_ratio(image, width=1280)

        # Stage 1: 预处理
        pre = self.preprocessor.process(image)

        # Stage 2: 定位
        loc = self.locator.locate(pre["color"], pre["gray"], method=self.locate_method)
        if loc["plate_region"] is None:
            result["error"] = "未检测到车牌区域"
            return result

        # 截取车牌区域 (带边距扩展)
        x, y, w, h = loc["plate_region"]["rect"]
        pad = max(w, h) // 10
        plate_image = image[max(0,y-pad):min(image.shape[0],y+h+pad),
                            max(0,x-pad):min(image.shape[1],x+w+pad)]

        # Stage 3: 倾斜校正
        corrected, skew_angle, _ = self.corrector.correct(plate_image, loc["plate_region"])

        # Stage 4: 字符分割
        seg = self.segmenter.segment(corrected, method=self.segment_method)

        # Stage 5: 字符识别
        if seg["chars"]:
            plate_text, char_details, avg_conf, _ = self.recognizer.recognize_all(seg["chars"])
        else:
            plate_text, char_details, avg_conf = "", [], 0.0

        # 结果汇总
        result.update({
            "success": True, "plate_text": plate_text,
            "confidence": avg_conf,
            "is_valid": validate_plate_format(plate_text),
            "plate_image": plate_image, "corrected_image": corrected,
            "char_images": seg["chars"],
            "result_image": draw_results(image, (x,y,w,h), plate_text),
            "total_time": (time.perf_counter() - total_start) * 1000,
        })
        return result

scripts/prepare_ccpd.py (核心逻辑)from lpr.utils import parse_ccpd_filename
from lpr.char_segmentation import CharSegmenter

def crop_plate_from_ccpd(image, annotation):
    """使用 CCPD 四角点标注透视变换裁剪车牌"""
    corners = np.array(annotation["corners"], dtype="float32")
    # 排序: 左上、右上、右下、左下
    s = corners.sum(axis=1)
    ordered = np.zeros((4, 2), dtype="float32")
    ordered[0] = corners[np.argmin(s)]   # 左上
    ordered[2] = corners[np.argmax(s)]   # 右下
    diff = np.diff(corners, axis=1)
    ordered[1] = corners[np.argmin(diff)] # 右上
    ordered[3] = corners[np.argmax(diff)] # 左下

    # 透视变换到 140x40 标准尺寸
    dst = np.array([[0,0],[139,0],[139,39],[0,39]], dtype="float32")
    M = cv2.getPerspectiveTransform(ordered, dst)
    return cv2.warpPerspective(image, M, (140, 40))


def prepare_ccpd(ccpd_dir, output_dir, max_samples=None):
    """主函数: 遍历 CCPD -> 裁剪 -> 分割 -> 保存"""
    segmenter = CharSegmenter()

    for img_path in all_images:
        # 1. 解析文件名
        annotation = parse_ccpd_filename(img_path.name)
        if annotation is None: continue

        # 2. 读取并裁剪
        image = cv2.imread(str(img_path))
        plate = crop_plate_from_ccpd(image, annotation)

        # 3. 分割字符
        result = segmenter.segment(plate, method="projection")
        if len(result["chars"]) != len(annotation["plate_text"]):
            continue  # 分割数量不匹配，跳过

        # 4. 保存: output_dir/train//.png
        for char_img, true_char in zip(result["chars"], annotation["plate_text"]):
            char_dir = output_dir / "train" / true_char
            char_dir.mkdir(parents=True, exist_ok=True)
            cv2.imwrite(str(char_dir / f"{idx}.png"), char_img)

常见问题

Q: 模板匹配准确率太低怎么办？
A: 合成模板只是演示用的。请使用 generate_templates.py --method ccpd 从 CCPD 数据集提取真实字符模板，准确率会显著提升。
Q: 车牌定位失败 (未检测到车牌区域)？
A: 1) 尝试切换定位方法为 multi (多颜色联合)；2) 调整 config.py 中的面积/宽高比阈值；3) 确认图片中车牌占比不要太小。
Q: 字符分割数量不对 (不是 7 个)？
A: 1) 调整 projection_threshold_ratio (降低会分割出更多区域)；2) 尝试连通域法 --segment_method connected；3) 检查倾斜校正是否生效。
Q: CNN 训练报错 "No module named torch"？
A: 需要单独安装 PyTorch: pip install torch torchvision。如果只使用模板匹配或 SVM，不需要安装 PyTorch。
Q: SVM 模型加载失败？
A: SVM 模型需要在 char_recognition.py 中正确加载。注意 config.py 中的 svm_model_path 路径要正确指向 models/svm_model.pkl。
